# Code Smell Annotation

- **Smell name:** Comments
- **Expected smell location:** `AuthManager` module, function `verify_token/2`
- **Affected function(s):** `verify_token/2`
- **Short explanation:** The function is documented using plain `#` comment blocks rather than `@doc`, which means the documentation is unavailable through ExDoc or the IEx `h/1` helper.

```elixir
defmodule AuthManager do
  @moduledoc """
  Handles token-based authentication and session management for the platform.
  """

  alias AuthManager.{Session, TokenStore, User}

  @token_ttl_seconds 3_600
  @refresh_window_seconds 300

  @doc """
  Issues a signed JWT for the given user, storing the session in the token store.
  Returns `{:ok, token}` on success or `{:error, reason}` on failure.
  """
  def issue_token(%User{id: user_id, roles: roles}) do
    claims = %{
      sub: user_id,
      roles: roles,
      iat: DateTime.utc_now() |> DateTime.to_unix(),
      exp: (DateTime.utc_now() |> DateTime.to_unix()) + @token_ttl_seconds
    }

    with {:ok, token} <- TokenStore.sign(claims),
         :ok <- TokenStore.persist(user_id, token) do
      {:ok, token}
    else
      {:error, reason} -> {:error, reason}
    end
  end

  # VALIDATION: SMELL START - Comments
  # VALIDATION: This is a smell because `verify_token/2` is documented using plain `#` comments
  # VALIDATION: instead of `@doc`, so the documentation is stripped at compile time and
  # VALIDATION: cannot be retrieved via IEx `h/1` or generated by ExDoc.

  # Verifies the authenticity and expiry of a given token string.
  # - token: binary string representing the signed JWT
  # - opts: keyword list of options
  #     :allow_expired  (boolean) - when true, skips expiry validation (default: false)
  #     :required_roles (list)    - list of role atoms the token must include
  # Returns {:ok, claims} if valid, {:error, :expired} if past expiry,
  # or {:error, :invalid} if the signature does not match.
  def verify_token(token, opts \\ []) when is_binary(token) do
    allow_expired = Keyword.get(opts, :allow_expired, false)
    required_roles = Keyword.get(opts, :required_roles, [])

    with {:ok, claims} <- TokenStore.verify_signature(token),
         :ok <- check_expiry(claims, allow_expired),
         :ok <- check_roles(claims, required_roles) do
      {:ok, claims}
    end
  end

  # VALIDATION: SMELL END

  @doc """
  Revokes an active session for the given user, invalidating the stored token.
  """
  def revoke_session(user_id) when is_binary(user_id) do
    case TokenStore.delete(user_id) do
      :ok -> {:ok, :revoked}
      {:error, :not_found} -> {:error, :session_not_found}
    end
  end

  @doc """
  Refreshes a token that is within the refresh window before expiry.
  Returns `{:ok, new_token}` or `{:error, reason}`.
  """
  def refresh_token(token) when is_binary(token) do
    with {:ok, claims} <- verify_token(token, allow_expired: true),
         :ok <- check_refresh_window(claims),
         %User{} = user <- TokenStore.load_user(claims.sub) do
      issue_token(user)
    else
      {:error, reason} -> {:error, reason}
    end
  end

  @doc """
  Returns the current session information for a user if an active token exists.
  """
  def get_session(user_id) when is_binary(user_id) do
    case TokenStore.fetch(user_id) do
      {:ok, token} -> {:ok, %Session{user_id: user_id, token: token}}
      {:error, :not_found} -> {:error, :no_active_session}
    end
  end

  defp check_expiry(%{exp: exp}, false) do
    now = DateTime.utc_now() |> DateTime.to_unix()
    if now > exp, do: {:error, :expired}, else: :ok
  end

  defp check_expiry(_claims, true), do: :ok

  defp check_roles(%{roles: token_roles}, required) do
    if Enum.all?(required, &(&1 in token_roles)),
      do: :ok,
      else: {:error, :insufficient_roles}
  end

  defp check_refresh_window(%{exp: exp}) do
    now = DateTime.utc_now() |> DateTime.to_unix()
    if exp - now <= @refresh_window_seconds, do: :ok, else: {:error, :not_in_refresh_window}
  end
end
```
