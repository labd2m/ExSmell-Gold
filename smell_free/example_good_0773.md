defmodule Search.SpellingCorrector do
  @moduledoc """
  Suggests spelling corrections for query terms using a pre-built word
  frequency dictionary. Corrections are generated by evaluating candidate
  words within edit distance 1 or 2 and ranking by dictionary frequency.
  The dictionary is loaded from application configuration at startup and
  cached in a persistent term for zero-latency lookups on the hot path.
  """

  @type word :: String.t()
  @type suggestion :: %{word: word(), score: non_neg_integer()}

  @alphabet String.graphemes("abcdefghijklmnopqrstuvwxyz")
  @max_edit_distance 2
  @max_suggestions 5

  @doc """
  Returns up to #{@max_suggestions} spelling suggestions for `word`.
  Returns an empty list when `word` is already in the dictionary or
  has no plausible corrections.
  """
  @spec suggest(word()) :: [suggestion()]
  def suggest(word) when is_binary(word) do
    lower = String.downcase(String.trim(word))
    dict = dictionary()

    if Map.has_key?(dict, lower) do
      []
    else
      candidates =
        lower
        |> edits1()
        |> MapSet.new()
        |> then(&expand_if_sparse(&1, lower))
        |> Enum.filter(&Map.has_key?(dict, &1))
        |> Enum.map(fn w -> %{word: w, score: Map.fetch!(dict, w)} end)
        |> Enum.sort_by(& &1.score, :desc)
        |> Enum.take(@max_suggestions)

      candidates
    end
  end

  @doc "Returns true when `word` exists in the configured dictionary."
  @spec known?(word()) :: boolean()
  def known?(word) when is_binary(word) do
    Map.has_key?(dictionary(), String.downcase(String.trim(word)))
  end

  @doc "Returns the frequency score for `word`, or 0 if not in the dictionary."
  @spec frequency(word()) :: non_neg_integer()
  def frequency(word) when is_binary(word) do
    Map.get(dictionary(), String.downcase(word), 0)
  end

  defp edits1(word) do
    splits = for i <- 0..String.length(word) do
      {String.slice(word, 0, i), String.slice(word, i, String.length(word))}
    end

    deletes = for {l, r} <- splits, r != "", do: l <> String.slice(r, 1, String.length(r))
    transposes = for {l, r} <- splits, String.length(r) >= 2 do
      l <> String.at(r, 1) <> String.at(r, 0) <> String.slice(r, 2, String.length(r))
    end
    replaces = for {l, r} <- splits, r != "", c <- @alphabet, do: l <> c <> String.slice(r, 1, String.length(r))
    inserts = for {l, r} <- splits, c <- @alphabet, do: l <> c <> r

    MapSet.new(deletes ++ transposes ++ replaces ++ inserts)
  end

  defp expand_if_sparse(candidates, word) do
    if @max_edit_distance >= 2 and MapSet.size(candidates) < 3 do
      candidates
      |> Enum.flat_map(&MapSet.to_list(edits1(&1)))
      |> MapSet.new()
      |> MapSet.union(candidates)
    else
      candidates
    end
  end

  defp dictionary do
    case :persistent_term.get({__MODULE__, :dict}, :miss) do
      :miss ->
        dict = Application.get_env(:my_app, :spelling_dictionary, %{})
        :persistent_term.put({__MODULE__, :dict}, dict)
        dict

      dict ->
        dict
    end
  end
end
