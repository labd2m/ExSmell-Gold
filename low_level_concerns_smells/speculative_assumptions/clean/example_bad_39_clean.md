defmodule HR.EmployeeImporter do
  @moduledoc """
  Imports employee records from TSV (tab-separated values) export files
  generated by the legacy HR system during monthly data migrations.

  Expected column order (0-indexed):
    0: employee_id
    1: first_name
    2: last_name
    3: email
    4: department
    5: job_title
    6: manager_id
    7: hire_date
    8: salary_band
    9: status
  """

  require Logger

  @expected_column_count 10
  @valid_statuses ~w(active inactive on_leave terminated)

  def import_file(path) do
    path
    |> File.stream!()
    |> Stream.drop(1)
    |> Stream.map(&String.trim_trailing(&1, "\n"))
    |> Stream.reject(&(&1 == ""))
    |> Stream.map(&build_record/1)
    |> Stream.reject(&match?({:error, _}, &1))
    |> Stream.map(fn {:ok, record} -> record end)
    |> Enum.to_list()
  end

  defp build_record(line) do
    cols = String.split(line, "\t")

    employee_id = Enum.at(cols, 0)
    first_name  = Enum.at(cols, 1)
    last_name   = Enum.at(cols, 2)
    email       = Enum.at(cols, 3)
    department  = Enum.at(cols, 4)
    job_title   = Enum.at(cols, 5)
    manager_id  = Enum.at(cols, 6)
    hire_date   = cols |> Enum.at(7) |> parse_date()
    salary_band = Enum.at(cols, 8)
    status      = Enum.at(cols, 9)

    record = %{
      employee_id: employee_id,
      first_name:  first_name,
      last_name:   last_name,
      email:       email,
      department:  department,
      job_title:   job_title,
      manager_id:  manager_id,
      hire_date:   hire_date,
      salary_band: salary_band,
      status:      status
    }

    {:ok, record}
  end

  def validate_record(%{employee_id: eid, email: email, status: status} = record) do
    cond do
      is_nil(eid) or eid == ""         -> {:error, :missing_employee_id}
      is_nil(email) or !valid_email?(email) -> {:error, :invalid_email}
      status not in @valid_statuses    -> {:error, {:invalid_status, status}}
      true                             -> {:ok, record}
    end
  end

  def validate_record(_), do: {:error, :malformed_record}

  defp valid_email?(email) do
    String.contains?(email, "@") and String.contains?(email, ".")
  end

  defp parse_date(nil), do: nil
  defp parse_date(""), do: nil
  defp parse_date(str) do
    case Date.from_iso8601(String.trim(str)) do
      {:ok, d} -> d
      _        -> nil
    end
  end

  def group_by_department(records) do
    Enum.group_by(records, & &1.department)
  end

  def active_employees(records) do
    Enum.filter(records, &(&1.status == "active"))
  end

  def headcount_report(records) do
    records
    |> Enum.group_by(& &1.department)
    |> Enum.map(fn {dept, emps} ->
      active = Enum.count(emps, &(&1.status == "active"))
      %{department: dept, total: length(emps), active: active}
    end)
    |> Enum.sort_by(& &1.total, :desc)
  end

  def full_name(%{first_name: first, last_name: last}), do: "#{first} #{last}"
  def full_name(_), do: "Unknown"
end
