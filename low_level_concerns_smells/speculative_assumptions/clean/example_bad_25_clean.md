defmodule Billing.DiscountCodeParser do
  @moduledoc """
  Decodes structured discount codes used in the promotions and billing system.

  Discount codes are generated by the marketing platform and embed the discount
  mechanics directly in the code string. This allows the billing service to
  apply discounts without a database lookup for simple, non-personalised codes.

  Format:
    "<TYPE>-<VALUE>-<SCOPE>"

  Types:
    PCT   — Percentage discount (value is an integer 1–100)
    FLAT  — Flat amount off in cents (value is a positive integer)
    FREE  — Free period in days (value is a positive integer)

  Scopes:
    ALL       — Applies to the entire invoice
    SAAS      — Applies to SaaS seat charges only
    SUPPORT   — Applies to support plan charges only
    FIRSTTERM — Applies to the first billing period only

  Examples:
    "PCT-20-ALL"
    "FLAT-5000-SAAS"
    "FREE-30-FIRSTTERM"
    "PCT-15-SUPPORT"
  """

  require Logger

  @known_types  ~w(PCT FLAT FREE)
  @known_scopes ~w(ALL SAAS SUPPORT FIRSTTERM)

  defstruct [:type, :value, :scope, :raw]

  @doc """
  Decodes a discount code string into a `%DiscountCodeParser{}` struct.

  Returns `{:ok, struct}` when type, value, and scope are valid.
  Returns `{:error, reason}` when any component fails validation.
  """

  def decode(code) when is_binary(code) do
    parts      = String.split(code, "-")
    type       = Enum.at(parts, 0)
    raw_value  = Enum.at(parts, 1)
    scope      = Enum.at(parts, 2)

    with :ok          <- validate_type(type),
         {:ok, value} <- parse_value(type, raw_value),
         :ok          <- validate_scope(scope) do
      {:ok, %__MODULE__{
        type:  type,
        value: value,
        scope: scope,
        raw:   code
      }}
    end
  end

  @doc """
  Applies a decoded discount to a line item amount (in cents).

  Returns the discounted amount in cents.
  """
  def apply_discount(%__MODULE__{type: "PCT",  value: pct},  amount_cents) do
    round(amount_cents * (1 - pct / 100))
  end

  def apply_discount(%__MODULE__{type: "FLAT", value: flat}, amount_cents) do
    max(0, amount_cents - flat)
  end

  def apply_discount(%__MODULE__{type: "FREE"}, _amount_cents) do
    0
  end

  @doc """
  Returns a human-readable summary of the discount for display in invoices.
  """
  def describe(%__MODULE__{type: "PCT",  value: pct,  scope: scope}) do
    "#{pct}% off #{humanize_scope(scope)}"
  end

  def describe(%__MODULE__{type: "FLAT", value: cents, scope: scope}) do
    dollars = Float.round(cents / 100, 2)
    "$#{dollars} off #{humanize_scope(scope)}"
  end

  def describe(%__MODULE__{type: "FREE", value: days}) do
    "#{days} days free"
  end

  @doc """
  Returns true when the discount applies to the full invoice.
  """
  def applies_to_all?(%__MODULE__{scope: "ALL"}), do: true
  def applies_to_all?(_), do: false

  @doc """
  Returns true when the discount is limited to the first billing period.
  """
  def first_term_only?(%__MODULE__{scope: "FIRSTTERM"}), do: true
  def first_term_only?(_), do: false

  # ---------------------------------------------------------------------------
  # Private helpers
  # ---------------------------------------------------------------------------

  defp validate_type(t) when is_binary(t) do
    if t in @known_types, do: :ok, else: {:error, {:unknown_discount_type, t}}
  end

  defp validate_type(nil), do: {:error, :missing_type}
  defp validate_type(_),   do: {:error, :invalid_type}

  defp validate_scope(s) when is_binary(s) do
    if s in @known_scopes, do: :ok, else: {:error, {:unknown_scope, s}}
  end

  defp validate_scope(nil), do: {:error, :missing_scope}
  defp validate_scope(_),   do: {:error, :invalid_scope}

  defp parse_value("PCT", str) when is_binary(str) do
    case Integer.parse(str) do
      {n, ""} when n >= 1 and n <= 100 -> {:ok, n}
      {n, ""}                           -> {:error, {:pct_out_of_range, n}}
      _                                 -> {:error, {:invalid_pct_value, str}}
    end
  end

  defp parse_value(type, str) when type in ["FLAT", "FREE"] and is_binary(str) do
    case Integer.parse(str) do
      {n, ""} when n > 0 -> {:ok, n}
      _                  -> {:error, {:invalid_value, str}}
    end
  end

  defp parse_value(_, nil), do: {:error, :missing_value}
  defp parse_value(_, _),   do: {:error, :invalid_value}

  defp humanize_scope("ALL"),       do: "the entire invoice"
  defp humanize_scope("SAAS"),      do: "SaaS charges"
  defp humanize_scope("SUPPORT"),   do: "support plan charges"
  defp humanize_scope("FIRSTTERM"), do: "first billing period"
  defp humanize_scope(other),       do: other
end
