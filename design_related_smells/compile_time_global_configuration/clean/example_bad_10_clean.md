defmodule Auth.OAuthProvider do
  @moduledoc """
  Implements the OAuth 2.0 Authorization Code flow against an external
  identity provider. Supports both initial code exchange and token refresh.
  Tokens are returned as structured maps and should be stored securely by
  the caller.
  """

  require Logger

  @oauth_authorize_url "https://sso.example.com/oauth/authorize"
  @oauth_token_url "https://sso.example.com/oauth/token"
  @oauth_client_secret Application.fetch_env!(:auth, :oauth_client_secret)

  @oauth_client_id Application.fetch_env!(:auth, :oauth_client_id)

  @default_scopes ~w(openid profile email)
  @token_request_timeout 10_000

  @type token_set :: %{
          access_token: String.t(),
          refresh_token: String.t() | nil,
          expires_in: pos_integer(),
          token_type: String.t()
        }

  @doc """
  Builds the URL to redirect an end-user to for authorization.

  ## Parameters
    - `redirect_uri` - The registered callback URI.
    - `state` - An opaque CSRF-protection value generated by the caller.
    - `scopes` - List of requested scopes; defaults to `#{inspect(@default_scopes)}`.
  """
  @spec authorization_url(String.t(), String.t(), [String.t()]) :: String.t()
  def authorization_url(redirect_uri, state, scopes \\ @default_scopes) do
    params = %{
      response_type: "code",
      client_id: @oauth_client_id,
      redirect_uri: redirect_uri,
      scope: Enum.join(scopes, " "),
      state: state
    }

    query = URI.encode_query(params)
    "#{@oauth_authorize_url}?#{query}"
  end

  @doc """
  Exchanges an authorization `code` for an access token set.

  ## Parameters
    - `code` - The short-lived authorization code returned by the provider.
    - `redirect_uri` - Must match the redirect URI used in the initial request.
  """
  @spec exchange_code(String.t(), String.t()) :: {:ok, token_set()} | {:error, term()}
  def exchange_code(code, redirect_uri) when is_binary(code) and is_binary(redirect_uri) do
    Logger.info("Exchanging OAuth code for tokens redirect_uri=#{redirect_uri}")

    params = %{
      grant_type: "authorization_code",
      code: code,
      redirect_uri: redirect_uri,
      client_id: @oauth_client_id,
      client_secret: @oauth_client_secret
    }

    post_token(params)
  end

  @doc """
  Uses a `refresh_token` to obtain a new access token without requiring user
  interaction.

  ## Parameters
    - `refresh_token` - A valid refresh token from a prior `exchange_code/2`.
  """
  @spec refresh_token(String.t()) :: {:ok, token_set()} | {:error, term()}
  def refresh_token(refresh_token) when is_binary(refresh_token) do
    Logger.info("Refreshing OAuth access token")

    params = %{
      grant_type: "refresh_token",
      refresh_token: refresh_token,
      client_id: @oauth_client_id,
      client_secret: @oauth_client_secret
    }

    post_token(params)
  end

  # ---------------------------------------------------------------------------
  # Private helpers
  # ---------------------------------------------------------------------------

  defp post_token(params) do
    body = URI.encode_query(params)

    headers = [
      {"Content-Type", "application/x-www-form-urlencoded"},
      {"Accept", "application/json"}
    ]

    case HTTPoison.post(@oauth_token_url, body, headers, recv_timeout: @token_request_timeout) do
      {:ok, %HTTPoison.Response{status_code: 200, body: resp_body}} ->
        token_set = resp_body |> Jason.decode!() |> normalise_token_set()
        {:ok, token_set}

      {:ok, %HTTPoison.Response{status_code: status, body: resp_body}} ->
        error = Jason.decode!(resp_body)
        Logger.error("OAuth token request failed status=#{status} error=#{inspect(error)}")
        {:error, error}

      {:error, %HTTPoison.Error{reason: reason}} ->
        Logger.error("HTTP error during OAuth token request reason=#{inspect(reason)}")
        {:error, %{message: inspect(reason)}}
    end
  end

  defp normalise_token_set(raw) do
    %{
      access_token: raw["access_token"],
      refresh_token: raw["refresh_token"],
      expires_in: raw["expires_in"],
      token_type: raw["token_type"] || "Bearer"
    }
  end
end
